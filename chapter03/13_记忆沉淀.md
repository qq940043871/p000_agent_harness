# 13 记忆沉淀：状态外部化，基于文件系统的持久化记忆与待办管理

## 🎯 本节目标

实现 Agent 的持久化记忆系统，让 Agent 能够跨任务积累知识，并通过文件系统管理待办事项。

---

## 1. 为什么 Agent 需要持久化记忆？

每次会话后 LLM 的上下文都会清空。没有持久化记忆的 Agent 就像患了失忆症：

```
第一天：用户告诉 Agent：数据库密码是 db_prod_xxx，避免用 ORM，直接用 psycopg2
第二天：Agent 还是用 SQLAlchemy，还要用户再说一遍密码...

有了持久化记忆：
MEMORY.md 记录了这些偏好 → Agent 下次启动时读取 → 一次都不用再说
```

## 2. 记忆层级设计

```
┌─────────────────────────────────────────┐
│           记忆层级（从短到长）            │
│                                          │
│  L1: 当前 Session 消息（最短，任务级）    │  见第 11 节

│  L3: 每日日志（长期，归档用）            │  本节实现
└─────────────────────────────────────────┘
```

## 4. 记忆管理工具实现

```python
# harness/memory/manager.py

import json
import re
import time
from datetime import datetime
from pathlib import Path
from typing import Optional


class MemoryManager:
    """
    持久化记忆管理器
    
    工具：
    - remember：将关键信息写入 MEMORY.md
    - recall：读取记忆内容
    - forget：删除指定记忆
    - daily_log：追加到每日日志
    """
    
    def __init__(self, workspace: str = "."):
        self.memory_dir = Path(workspace) / ".workbuddy" / "memory"
        self.memory_dir.mkdir(parents=True, exist_ok=True)
        self.memory_file = self.memory_dir / "MEMORY.md"
    
    def remember(self, section: str, content: str) -> str:
        """
        将信息写入 MEMORY.md 的指定章节
        
        Args:
            section: 章节名（如"用户偏好"、"项目约定"）
            content: 要记住的内容
        """
        if not self.memory_file.exists():
            self._init_memory_file()
        
        memory = self.memory_file.read_text(encoding="utf-8")
        
        # 查找或创建章节
        section_pattern = rf"## {re.escape(section)}\n"
        if re.search(section_pattern, memory):
            # 章节存在，在章节内容末尾追加
            def add_to_section(match):
                return match.group(0) + f"- {content}\n"
            memory = re.sub(
                rf"(## {re.escape(section)}\n(?:.*\n)*?)(?=## |\Z)",
                lambda m: m.group(0).rstrip() + f"\n- {content}\n\n",
                memory,
                count=1,
                flags=re.MULTILINE,
            )
        else:
            # 章节不存在，在文件末尾新建
            memory = memory.rstrip() + f"\n\n## {section}\n- {content}\n"
        
        # 更新最后修改时间
        today = datetime.now().strftime("%Y-%m-%d")
        memory = re.sub(r"_最后更新: .*_", f"_最后更新: {today}_", memory)
        
        self.memory_file.write_text(memory, encoding="utf-8")
        return f"✓ 已记录到「{section}」章节"
    
    def recall(self, query: str = None) -> str:
        """
        读取记忆内容
        
        Args:
            query: 如果指定，返回相关章节；否则返回全部
        """
        if not self.memory_file.exists():
            return "（记忆库为空）"
        
        content = self.memory_file.read_text(encoding="utf-8")
        
        if not query:
            return content
        
        # 简单的关键词匹配
        relevant_lines = [
            line for line in content.split("\n")
            if query.lower() in line.lower()
        ]
        
        return "\n".join(relevant_lines) if relevant_lines else f"未找到与「{query}」相关的记忆"
    
    def daily_log(self, entry: str) -> str:
        """
        追加一条记录到今天的日志文件
        
        日志文件：.workbuddy/memory/YYYY-MM-DD.md
        """
        today = datetime.now().strftime("%Y-%m-%d")
        log_file = self.memory_dir / f"{today}.md"
        
        if not log_file.exists():
            log_file.write_text(f"# 工作日志 {today}\n\n", encoding="utf-8")
        
        timestamp = datetime.now().strftime("%H:%M")
        with open(log_file, "a", encoding="utf-8") as f:
            f.write(f"- [{timestamp}] {entry}\n")
        
        return f"✓ 已记录到今日日志"
    
    def _init_memory_file(self):
        """初始化空的 MEMORY.md"""
        today = datetime.now().strftime("%Y-%m-%d")
        self.memory_file.write_text(
            f"# 持久化记忆\n\n_最后更新: {today}_\n\n",
            encoding="utf-8",
        )
```

## 5. 待办管理工具

```python
# harness/memory/todos.py

import json
from datetime import datetime
from pathlib import Path
from dataclasses import dataclass, asdict
from typing import Optional
from enum import Enum


class TodoStatus(str, Enum):
    PENDING = "pending"
    IN_PROGRESS = "in_progress"
    COMPLETED = "completed"
    CANCELLED = "cancelled"


@dataclass
class Todo:
    id: str
    content: str
    status: TodoStatus = TodoStatus.PENDING
    created_at: str = ""
    completed_at: str = ""
    priority: int = 0  # 0=普通, 1=重要, 2=紧急


class TodoManager:
    """基于文件系统的待办管理器"""
    
    def __init__(self, workspace: str = "."):
        self.todo_file = Path(workspace) / ".workbuddy" / "todos.json"
        self.todo_file.parent.mkdir(parents=True, exist_ok=True)
    
    def _load(self) -> list[dict]:
        if self.todo_file.exists():
            return json.loads(self.todo_file.read_text())
        return []
    
    def _save(self, todos: list[dict]):
        self.todo_file.write_text(
            json.dumps(todos, indent=2, ensure_ascii=False)
        )
    
    def add_todo(self, content: str, priority: int = 0) -> str:
        todos = self._load()
        todo_id = f"todo_{len(todos) + 1:04d}"
        todos.append(asdict(Todo(
            id=todo_id,
            content=content,
            created_at=datetime.now().isoformat(),
            priority=priority,
        )))
        self._save(todos)
        return f"✓ 已添加待办 [{todo_id}]: {content}"
    
    def complete_todo(self, todo_id: str) -> str:
        todos = self._load()
        for todo in todos:
            if todo["id"] == todo_id:
                todo["status"] = TodoStatus.COMPLETED
                todo["completed_at"] = datetime.now().isoformat()
                self._save(todos)
                return f"✓ 已完成: {todo['content']}"
        return f"未找到待办 ID: {todo_id}"
    
    def list_todos(self, status: str = None) -> str:
        todos = self._load()
        if status:
            todos = [t for t in todos if t["status"] == status]
        
        if not todos:
            return "（无待办事项）"
        
        lines = []
        for todo in todos:
            emoji = {"pending": "⬜", "in_progress": "🔄", "completed": "✅", "cancelled": "❌"}.get(todo["status"], "?")
            lines.append(f"{emoji} [{todo['id']}] {todo['content']}")
        
        return "\n".join(lines)
```

## 6. 将记忆工具注册到 Agent

```python
memory_manager = MemoryManager(workspace)
todo_manager = TodoManager(workspace)

@registry.tool(description="将重要信息永久保存到记忆库，下次启动时会自动读取", category="memory")
async def remember(section: str, content: str) -> str:
    return memory_manager.remember(section, content)

@registry.tool(description="读取记忆库中的信息", category="memory")
async def recall(query: str = "") -> str:
    return memory_manager.recall(query or None)

@registry.tool(description="记录一条工作日志到今天的日志文件", category="memory")
async def daily_log(entry: str) -> str:
    return memory_manager.daily_log(entry)

@registry.tool(description="添加一个待办事项", category="todos")
async def add_todo(content: str) -> str:
    return todo_manager.add_todo(content)

@registry.tool(description="标记待办事项为已完成", category="todos")
async def complete_todo(todo_id: str) -> str:
    return todo_manager.complete_todo(todo_id)

@registry.tool(description="列出所有待办事项", category="todos")
async def list_todos(status: str = "") -> str:
    return todo_manager.list_todos(status or None)
```

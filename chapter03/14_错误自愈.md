# 14 错误自愈：上下文感知的 Error Recovery 提示模板注入机制

## 🎯 本节目标

实现智能的错误恢复机制，当 Agent 遇到工具执行失败时，自动注入上下文感知的恢复提示，引导 Agent 自主修复问题。

---

## 1. Agent 错误的分类

```
错误类型
├── 工具执行错误
│   ├── 文件不存在（FileNotFoundError）
│   ├── 命令执行失败（非零退出码）
│   ├── 权限不足（PermissionError）
│   └── 超时（TimeoutError）
├── 逻辑错误
│   ├── 参数格式错误（JSON 解析失败）
│   ├── 工具名称幻觉（调用不存在的工具）
│   └── 循环依赖（A 等 B，B 等 A）
└── 上下文错误
    ├── 代码语法错误（Python SyntaxError）
    ├── 测试失败
    └── 依赖缺失（ModuleNotFoundError）
```

---

## 2. 朴素错误处理 vs 上下文感知

```python
# ❌ 朴素错误处理：直接将错误返回给 LLM
result = "Error: No such file or directory: '/app/config.py'"
# LLM 可能不知道该怎么处理，或者做出错误的推断

# ✓ 上下文感知：注入恢复提示
result = """
工具执行失败: No such file or directory: '/app/config.py'

[错误分析]
文件路径不存在。可能的原因：
1. 文件路径错误（拼写/大小写）
2. 文件尚未创建
3. 工作目录不正确

[建议恢复步骤]
- 使用 list_dir 查看当前目录结构，确认正确路径
- 或者使用 bash("find . -name 'config.py'") 搜索文件位置
"""
# LLM 得到了明确的指引，成功率大幅提升
```

## 3. Error Recovery 系统实现

```python
# harness/recovery/error_handler.py

import re
import traceback
from dataclasses import dataclass
from typing import Optional, Callable


@dataclass
class ErrorContext:
    """错误上下文信息"""
    error_type: str
    error_message: str
    tool_name: str
    tool_args: dict
    raw_exception: Optional[Exception] = None


class ErrorRecoverySystem:
    """
    上下文感知的错误恢复系统
    
    根据错误类型自动生成恢复提示
    """
    
    def __init__(self):
        self._handlers: list[tuple[Callable, Callable]] = []
        self._register_default_handlers()
    
    def register_handler(
        self,
        matcher: Callable[[ErrorContext], bool],
        generator: Callable[[ErrorContext], str],
    ):
        """注册自定义错误处理器"""
        self._handlers.append((matcher, generator))
    
    def get_recovery_hint(self, ctx: ErrorContext) -> str:
        """根据错误上下文生成恢复提示"""
        for matcher, generator in self._handlers:
            if matcher(ctx):
                hint = generator(ctx)
                return self._format_error_response(ctx, hint)
        
        # 默认处理
        return self._format_error_response(ctx, "请检查工具调用参数并重试。")
    
    def _format_error_response(self, ctx: ErrorContext, hint: str) -> str:
        return (
            f"工具执行失败: {ctx.error_message}\n\n"
            f"[错误分析与恢复建议]\n{hint}"
        )
    
    def _register_default_handlers(self):
        """注册默认的错误处理规则"""
        
        # ── 文件不存在 ─────────────────────────────────────
        def is_file_not_found(ctx: ErrorContext) -> bool:
            return (
                "No such file" in ctx.error_message or
                "FileNotFoundError" in ctx.error_type or
                "not found" in ctx.error_message.lower()
            )
        
        def handle_file_not_found(ctx: ErrorContext) -> str:
            path = ctx.tool_args.get("path", "")
            return (
                f"路径 '{path}' 不存在。\n\n"
                f"建议步骤：\n"
                f"1. 使用 `list_dir` 查看目录结构，确认正确路径\n"
                f"2. 使用 `bash(\"find . -name '{path.split('/')[-1']}\")` 搜索文件\n"
                f"3. 如需创建文件，使用 `write_file` 工具"
            )
        
        self.register_handler(is_file_not_found, handle_file_not_found)
        
        # ── 命令执行失败 ────────────────────────────────────
        def is_command_failure(ctx: ErrorContext) -> bool:
            return (
                ctx.tool_name == "bash" and
                "[Error]" in ctx.error_message
            )
        
        def handle_command_failure(ctx: ErrorContext) -> str:
            command = ctx.tool_args.get("command", "")
            stderr = ctx.error_message
            
            # 特殊情况：Python 语法错误
            if "SyntaxError" in stderr:
                return (
                    f"Python 语法错误。\n"
                    f"请检查代码语法，常见问题：\n"
                    f"- 缩进错误（混用 Tab 和空格）\n"
                    f"- 括号未闭合\n"
                    f"- 使用了 Python 2 语法\n\n"
                    f"建议：先用 `read_file` 查看文件当前内容，再修改"
                )
            
            # 特殊情况：模块未找到
            if "ModuleNotFoundError" in stderr or "No module named" in stderr:
                module = re.search(r"No module named '(\S+)'", stderr)
                module_name = module.group(1) if module else "未知"
                return (
                    f"缺少 Python 模块 '{module_name}'。\n"
                    f"解决方案：`bash(\"pip install {module_name}\")`"
                )
            
            # 特殊情况：权限不足
            if "Permission denied" in stderr:
                return (
                    f"权限不足，无法执行该命令。\n"
                    f"建议：\n"
                    f"- 确认是否需要 sudo（当前环境可能不允许）\n"
                    f"- 检查目标文件/目录的权限"
                )
            
            return (
                f"命令执行出错。\n"
                f"命令: `{command}`\n\n"
                f"建议：\n"
                f"- 检查命令语法\n"
                f"- 确认依赖工具已安装\n"
                f"- 简化命令逐步调试"
            )
        
        self.register_handler(is_command_failure, handle_command_failure)
        
        # ── 超时 ─────────────────────────────────────────
        def is_timeout(ctx: ErrorContext) -> bool:
            return "超时" in ctx.error_message or "Timeout" in ctx.error_type
        
        def handle_timeout(ctx: ErrorContext) -> str:
            return (
                f"工具执行超时（> 30s）。\n\n"
                f"可能原因：\n"
                f"- 命令需要用户交互输入\n"
                f"- 网络请求无响应\n"
                f"- 大文件处理耗时过长\n\n"
                f"建议：\n"
                f"- 为命令添加非交互参数（如 -y, --no-input）\n"
                f"- 拆分为多个小任务执行"
            )
        
        self.register_handler(is_timeout, handle_timeout)
        
        # ── 工具不存在（幻觉） ─────────────────────────────
        def is_tool_not_found(ctx: ErrorContext) -> bool:
            return "不存在" in ctx.error_message and "工具" in ctx.error_message
        
        def handle_tool_not_found(ctx: ErrorContext) -> str:
            available = ctx.error_message.split("可用工具: ")[-1] if "可用工具" in ctx.error_message else ""
            return (
                f"工具 '{ctx.tool_name}' 不在可用工具列表中。\n\n"
                f"可用工具：{available}\n\n"
                f"请从可用工具中选择合适的工具完成任务。"
            )
        
        self.register_handler(is_tool_not_found, handle_tool_not_found)
```

## 4. 连接到 ToolExecutor

```python
# 在 ToolExecutor 中集成错误恢复

class ToolExecutor:
    def __init__(self, registry, recovery_system: ErrorRecoverySystem = None, ...):
        self.recovery = recovery_system or ErrorRecoverySystem()
    
    async def _execute_single(self, call) -> ToolCallResult:
        start = time.time()
        try:
            result = await asyncio.wait_for(
                self.registry.dispatch(call.name, call.args),
                timeout=self.timeout,
            )
            return ToolCallResult(
                tool_call_id=call.id,
                name=call.name,
                result=result,
                elapsed=time.time() - start,
            )
        
        except Exception as e:
            ctx = ErrorContext(
                error_type=type(e).__name__,
                error_message=str(e),
                tool_name=call.name,
                tool_args=call.args,
                raw_exception=e,
            )
            # 生成上下文感知的恢复提示
            recovery_hint = self.recovery.get_recovery_hint(ctx)
            
            return ToolCallResult(
                tool_call_id=call.id,
                name=call.name,
                result=recovery_hint,  # 替换原始错误消息
                elapsed=time.time() - start,
            )
```

## 5. 错误恢复效果对比

```
场景：Agent 尝试读取不存在的文件

❌ 无错误恢复：
工具返回: "No such file or directory: '/app/config.py'"
→ LLM 可能：重试 → 再次失败 → 循环 3 次后放弃

✓ 有错误恢复：
工具返回: "路径 '/app/config.py' 不存在。
建议步骤：
1. 使用 list_dir 查看目录结构，确认正确路径
..."
→ LLM：执行 list_dir → 发现文件在 /app/configs/config.py → 一次修正
```

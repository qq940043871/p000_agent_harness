# 12 突破内存：基于阶梯降级的 Context Compaction 策略

## 🎯 本节目标

理解 LLM 上下文窗口限制的本质，掌握多级 Context Compaction 策略，让 Agent 能够处理超长任务。

---

## 1. 上下文窗口：Agent 的内存限制

所有 LLM 都有上下文窗口限制：

| 模型 | 上下文窗口 |
|---|---|
| Claude Opus 4.5 | 200K tokens |
| GPT-4o | 128K tokens |
| Qwen-Max | 32K tokens |

听起来很大，但一个真实的代码修复任务可能包含：
- System Prompt：~2K tokens
- 项目代码文件内容：~30K tokens
- 多轮对话历史：~10K tokens（每轮迭代）

**50 轮循环后，消耗就会逼近限制。**

---

## 2. 阶梯降级策略

OpenClaw 采用三级阶梯，窗口用量越高，压缩越激进：

```
Token 使用率
0%   ████████████████████  100%

0% - 60%：正常运行（无压缩）
60% - 80%：轻度压缩（移除工具结果详情）
80% - 90%：中度压缩（保留摘要，丢弃工具历史）
90% +：重度压缩（仅保留最近 N 轮 + 任务上下文）
```

---

## 3. Context Compaction 实现

```python
# harness/context/compaction.py

import logging
from dataclasses import dataclass
from typing import Optional

logger = logging.getLogger(__name__)


@dataclass
class CompactionConfig:
    """压缩配置"""
    light_threshold: float = 0.60    # 轻度压缩触发阈值
    medium_threshold: float = 0.80   # 中度压缩触发阈值
    heavy_threshold: float = 0.90    # 重度压缩触发阈值
    keep_recent_turns: int = 5       # 重度压缩时保留最近 N 轮
    max_tool_result_chars: int = 2000  # 轻度压缩：工具结果最大字符数


class ContextCompactor:
    """
    基于阶梯降级的上下文压缩器
    
    在 Main Loop 每轮开始前检查并压缩
    """
    
    def __init__(
        self,
        provider,
        max_context_tokens: int,
        config: CompactionConfig = None,
    ):
        self.provider = provider
        self.max_context_tokens = max_context_tokens
        self.config = config or CompactionConfig()
        self._compaction_count = 0
    
    async def maybe_compact(
        self,
        messages: list[dict],
        current_tokens: int,
    ) -> tuple[list[dict], str]:
        """
        根据当前 Token 使用率决定是否压缩
        
        Returns:
            (压缩后的消息列表, 压缩级别描述)
        """
        usage_ratio = current_tokens / self.max_context_tokens
        
        if usage_ratio < self.config.light_threshold:
            return messages, "none"
        
        elif usage_ratio < self.config.medium_threshold:
            return self._light_compact(messages), "light"
        
        elif usage_ratio < self.config.heavy_threshold:
            return self._medium_compact(messages), "medium"
        
        else:
            compacted = await self._heavy_compact(messages)
            return compacted, "heavy"
    
    def _light_compact(self, messages: list[dict]) -> list[dict]:
        """
        轻度压缩：截断过长的工具返回结果
        保留所有消息结构，只压缩内容
        """
        compacted = []
        for msg in messages:
            if msg.get("role") == "tool":
                content = msg.get("content", "")
                if len(content) > self.config.max_tool_result_chars:
                    truncated = content[:self.config.max_tool_result_chars]
                    msg = {**msg, "content": truncated + "\n[...内容已截断...]"}
            compacted.append(msg)
        
        logger.info(f"轻度压缩：截断工具结果")
        return compacted
    
    def _medium_compact(self, messages: list[dict]) -> list[dict]:
        """
        中度压缩：保留 System Prompt 和最近的用户/Assistant 交换
        将中间的工具调用历史合并为摘要
        """
        system_messages = [m for m in messages if m.get("role") == "system"]
        user_messages = [m for m in messages if m.get("role") == "user"]
        
        if not user_messages:
            return messages
        
        # 保留第一个用户消息（原始任务）
        first_user_msg = user_messages[0]
        
        # 统计中间发生了什么
        tool_calls_count = sum(
            1 for m in messages
            if m.get("role") == "assistant" and self._has_tool_calls(m)
        )
        
        summary_msg = {
            "role": "user",
            "content": (
                f"[上下文摘要] 任务进行中。已执行 {tool_calls_count} 次工具调用。"
                f"继续完成原始任务。"
            )
        }
        
        # 保留最近几条消息
        recent_messages = messages[-self.config.keep_recent_turns * 2:]
        
        compacted = system_messages + [first_user_msg, summary_msg] + recent_messages
        
        logger.info(
            f"中度压缩：{len(messages)} → {len(compacted)} 条消息，"
            f"保留最近 {len(recent_messages)} 条"
        )
        return compacted
    
    async def _heavy_compact(self, messages: list[dict]) -> list[dict]:
        """
        重度压缩：让 LLM 生成整个上下文的摘要
        
        这是最激进的策略，适用于窗口即将耗尽的情况
        """
        self._compaction_count += 1
        logger.warning(f"重度压缩（第 {self._compaction_count} 次）：召唤 LLM 生成摘要")
        
        system_messages = [m for m in messages if m.get("role") == "system"]
        non_system = [m for m in messages if m.get("role") != "system"]
        
        # 构建摘要请求
        summary_request = [
            *non_system[-50:],  # 最后 50 条（避免这个请求本身也超限）
            {
                "role": "user",
                "content": (
                    "请用简洁的语言总结上述对话的关键信息：\n"
                    "1. 原始任务是什么\n"
                    "2. 已经完成了哪些步骤\n"
                    "3. 当前状态如何\n"
                    "4. 还需要完成什么\n"
                    "请用 300 字以内完成总结。"
                )
            }
        ]
        
        try:
            summary_response = await self.provider.complete(
                messages=summary_request,
                tools=[],
                max_tokens=500,
            )
            summary_content = summary_response.content
        except Exception as e:
            logger.error(f"摘要生成失败: {e}")
            summary_content = "（摘要生成失败，继续执行）"
        
        # 重置为精简版历史
        compacted = system_messages + [
            {
                "role": "user",
                "content": f"[压缩历史摘要]\n{summary_content}\n\n请根据以上摘要继续完成任务。"
            }
        ]
        
        logger.info(f"重度压缩：{len(messages)} → {len(compacted)} 条消息")
        return compacted
    
    def _has_tool_calls(self, message: dict) -> bool:
        """检查 assistant 消息是否包含工具调用"""
        content = message.get("content", "")
        if isinstance(content, list):
            return any(
                block.get("type") in ("tool_use", "tool_calls")
                for block in content
                if isinstance(block, dict)
            )
        return "tool_calls" in message


# ── 估算 Token 数量（轻量级，不调用 API）──────────────────────

def estimate_tokens(messages: list[dict]) -> int:
    """
    快速估算消息列表的 Token 数量
    
    规则：约 4 个字符 = 1 个 Token（英文）
         约 2 个字符 = 1 个 Token（中文）
    这是粗略估计，实际误差约 ±20%
    """
    total_chars = 0
    for msg in messages:
        content = msg.get("content", "")
        if isinstance(content, str):
            total_chars += len(content)
        elif isinstance(content, list):
            for block in content:
                if isinstance(block, dict):
                    total_chars += len(str(block.get("text", "") or block.get("content", "")))
    
    # 中英文混合，取平均系数 3
    return total_chars // 3
```

## 4. 在 Main Loop 中集成

```python
class MainLoop:
    def __init__(self, ..., max_context_tokens: int = 150000):
        self.compactor = ContextCompactor(
            provider=provider,
            max_context_tokens=max_context_tokens,
        )
    
    async def run(self, user_message: str) -> LoopResult:
        ...
        for turn in range(self.max_turns):
            
            # 每轮开始前检查并压缩
            current_tokens = estimate_tokens(messages)
            messages, compact_level = await self.compactor.maybe_compact(
                messages, current_tokens
            )
            
            if compact_level != "none":
                logger.info(f"[Turn {turn+1}] 上下文压缩级别: {compact_level}")
            
            response = await self.provider.complete(messages, ...)
            ...
```

## 📝 本节小结

| 压缩级别 | 触发条件 | 策略 | 信息损失 |
|---|---|---|---|
| 无压缩 | < 60% | 不处理 | 0 |
| 轻度 | 60-80% | 截断长工具输出 | 微小 |
| 中度 | 80-90% | 保留摘要+最近轮次 | 中等 |
| 重度 | > 90% | LLM 生成全局摘要 | 较大 |

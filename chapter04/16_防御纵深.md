# 16 防御纵深：利用 Middleware 实现高危命令拦截与飞书人工审批

## 🎯 本节目标

构建 Middleware 拦截层，实现高危命令的自动识别与拦截，并集成飞书审批流程实现人工确认机制。

---

## 1. 防御纵深架构

```
工具调用请求
      ↓
┌─────────────────────────────────────┐
│         Middleware 链                │
│                                      │
│  [日志记录]    → 记录所有调用         │
│  [危险检测]    → 识别高危操作         │
│       ↓                              │
│  [自动拒绝]    → 明显危险直接拒绝     │
│  [人工审批]    → 可疑操作飞书审批     │
│  [速率限制]    → 防止资源滥用         │
│                                      │
└────────────────┬────────────────────┘
                 ↓
         工具实际执行
```

## 2. Middleware 基础架构

```python
# harness/middleware/base.py

from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Any, Optional
from enum import Enum


class MiddlewareDecision(Enum):
    ALLOW = "allow"           # 允许执行
    DENY = "deny"             # 拒绝执行
    REQUIRE_APPROVAL = "require_approval"  # 需要审批
    MODIFY = "modify"         # 修改后执行


@dataclass
class MiddlewareResult:
    """Middleware 处理结果"""
    decision: MiddlewareDecision
    message: str = ""
    modified_args: Optional[dict] = None


@dataclass
class ToolCallRequest:
    """待处理的工具调用请求"""
    tool_name: str
    tool_args: dict
    call_id: str
    session_id: str = ""


class BaseMiddleware(ABC):
    """Middleware 基类"""
    
    @abstractmethod
    async def process(self, request: ToolCallRequest) -> MiddlewareResult:
        """处理工具调用请求"""
        ...


class MiddlewareChain:
    """
    Middleware 责任链
    
    按顺序执行所有 Middleware，任一 DENY/REQUIRE_APPROVAL 即停止
    """
    
    def __init__(self, middlewares: list[BaseMiddleware] = None):
        self._middlewares = middlewares or []
    
    def add(self, middleware: BaseMiddleware) -> "MiddlewareChain":
        self._middlewares.append(middleware)
        return self
    
    async def execute(self, request: ToolCallRequest) -> MiddlewareResult:
        for middleware in self._middlewares:
            result = await middleware.process(request)
            if result.decision != MiddlewareDecision.ALLOW:
                return result
        return MiddlewareResult(decision=MiddlewareDecision.ALLOW)
```

## 3. 危险命令检测 Middleware

```python
# harness/middleware/danger_detector.py

import re
from .base import BaseMiddleware, MiddlewareResult, MiddlewareDecision, ToolCallRequest


# 立即拒绝的绝对高危模式
ABSOLUTE_DENY_PATTERNS = [
    r"rm\s+-rf\s+[/~]",           # rm -rf / 或 ~/
    r"rm\s+-rf\s+\.\s*$",         # rm -rf .（当前目录）
    r":\(\)\s*\{\s*:\|:\s*&\s*\}", # Fork Bomb
    r"dd\s+if=.*of=/dev/[sh]d",   # 覆盖磁盘
    r"mkfs\.",                      # 格式化分区
    r"> /dev/sda",                  # 清除磁盘
    r"chmod\s+-R\s+777\s+/",      # 全局权限修改
]

# 需要人工确认的高危模式
APPROVAL_REQUIRED_PATTERNS = [
    r"rm\s+-rf",                   # 任何 rm -rf
    r"DROP\s+TABLE",               # 数据库表删除
    r"DROP\s+DATABASE",            # 数据库删除
    r"DELETE\s+FROM\s+\w+\s*;",   # 无 WHERE 的 DELETE
    r"TRUNCATE\s+TABLE",           # 清空表
    r"git\s+push.*--force",        # 强制推送
    r"kubectl\s+delete\s+namespace", # 删除 K8s 命名空间
    r"aws\s+s3\s+rb",              # 删除 S3 bucket
]

# 高危文件路径
SENSITIVE_PATHS = [
    r"/etc/passwd",
    r"/etc/shadow",
    r"\.env",
    r"\.ssh/",
    r"id_rsa",
    r"private_key",
]


class DangerCommandDetector(BaseMiddleware):
    """危险命令检测 Middleware"""
    
    async def process(self, request: ToolCallRequest) -> MiddlewareResult:
        command = request.tool_args.get("command", "")
        path = request.tool_args.get("path", "")
        content = str(request.tool_args.get("content", ""))
        
        check_text = f"{command} {path} {content}"
        
        # 检查绝对拒绝模式
        for pattern in ABSOLUTE_DENY_PATTERNS:
            if re.search(pattern, check_text, re.IGNORECASE):
                return MiddlewareResult(
                    decision=MiddlewareDecision.DENY,
                    message=(
                        f"⛔ 操作被系统拦截：检测到极高危险命令。\n"
                        f"匹配模式: `{pattern}`\n"
                        f"该操作可能导致不可逆的系统损坏，已被永久拒绝。\n"
                        f"请重新思考你的操作策略。"
                    )
                )
        
        # 检查敏感文件路径
        for pattern in SENSITIVE_PATHS:
            if re.search(pattern, check_text, re.IGNORECASE):
                return MiddlewareResult(
                    decision=MiddlewareDecision.DENY,
                    message=(
                        f"⛔ 操作被拦截：涉及敏感文件路径。\n"
                        f"路径 `{path}` 属于系统敏感文件，不允许操作。"
                    )
                )
        
        # 检查需要审批的模式
        for pattern in APPROVAL_REQUIRED_PATTERNS:
            if re.search(pattern, check_text, re.IGNORECASE):
                return MiddlewareResult(
                    decision=MiddlewareDecision.REQUIRE_APPROVAL,
                    message=(
                        f"⚠️ 检测到高危操作，需要人工确认。\n"
                        f"操作类型: {request.tool_name}\n"
                        f"操作内容: {check_text[:200]}\n"
                        f"匹配模式: `{pattern}`"
                    )
                )
        
        return MiddlewareResult(decision=MiddlewareDecision.ALLOW)
```

## 4. 飞书审批 Middleware

```python
# harness/middleware/feishu_approval.py

import asyncio
import uuid
import time
from .base import BaseMiddleware, MiddlewareResult, MiddlewareDecision, ToolCallRequest


class FeishuApprovalMiddleware(BaseMiddleware):
    """
    飞书人工审批 Middleware
    
    高危操作触发审批时，向指定飞书用户发送审批请求，
    等待用户确认（同意/拒绝）后继续执行。
    """
    
    def __init__(
        self,
        feishu_bot,                    # FeishuBot 实例
        approver_open_id: str,         # 审批人的飞书 open_id
        timeout: float = 300.0,        # 审批超时（秒）
    ):
        self.bot = feishu_bot
        self.approver = approver_open_id
        self.timeout = timeout
        self._pending_approvals: dict[str, asyncio.Event] = {}
        self._approval_results: dict[str, bool] = {}
    
    async def process(self, request: ToolCallRequest) -> MiddlewareResult:
        """不直接触发审批，等待 DangerCommandDetector 标记后处理"""
        # 这个 Middleware 配合 DangerCommandDetector 使用
        # 在 MiddlewareChain 中，前者标记需要审批，后者执行审批
        return MiddlewareResult(decision=MiddlewareDecision.ALLOW)
    
    async def request_approval(
        self,
        request: ToolCallRequest,
        reason: str,
    ) -> bool:
        """
        向飞书审批人发送审批请求，等待响应
        
        Returns:
            True = 批准, False = 拒绝/超时
        """
        approval_id = str(uuid.uuid4())[:8]
        event = asyncio.Event()
        self._pending_approvals[approval_id] = event
        
        # 发送审批卡片消息
        await self.bot.send_message(
            receive_id=self.approver,
            content=self._build_approval_card(approval_id, request, reason),
            receive_id_type="open_id",
        )
        
        # 等待审批结果
        try:
            await asyncio.wait_for(event.wait(), timeout=self.timeout)
            approved = self._approval_results.get(approval_id, False)
        except asyncio.TimeoutError:
            approved = False
            result = "⏰ 审批超时，操作已自动拒绝"
        finally:
            self._pending_approvals.pop(approval_id, None)
            self._approval_results.pop(approval_id, None)
        
        return approved
    
    def handle_approval_response(self, approval_id: str, approved: bool):
        """处理飞书回调的审批结果"""
        self._approval_results[approval_id] = approved
        if approval_id in self._pending_approvals:
            self._pending_approvals[approval_id].set()
    
    def _build_approval_card(
        self,
        approval_id: str,
        request: ToolCallRequest,
        reason: str,
    ) -> str:
        import json
        card = {
            "config": {"wide_screen_mode": True},
            "header": {
                "title": {"tag": "plain_text", "content": "⚠️ Agent 操作审批"},
                "template": "orange",
            },
            "elements": [
                {"tag": "div", "text": {"tag": "lark_md", "content": f"**操作类型:** `{request.tool_name}`"}},
                {"tag": "div", "text": {"tag": "lark_md", "content": f"**危险原因:** {reason}"}},
                {"tag": "div", "text": {"tag": "lark_md", "content": f"**操作参数:**\n```\n{json.dumps(request.tool_args, indent=2, ensure_ascii=False)}\n```"}},
                {"tag": "hr"},
                {
                    "tag": "action",
                    "actions": [
                        {
                            "tag": "button",
                            "text": {"tag": "plain_text", "content": "✅ 批准"},
                            "type": "primary",
                            "value": {"approval_id": approval_id, "approved": "true"},
                        },
                        {
                            "tag": "button",
                            "text": {"tag": "plain_text", "content": "❌ 拒绝"},
                            "type": "danger",
                            "value": {"approval_id": approval_id, "approved": "false"},
                        },
                    ]
                }
            ]
        }
        return json.dumps(card)
```

## 5. 完整的 Middleware Chain 配置

```python
# 构建防御纵深 Middleware 链

from harness.middleware import MiddlewareChain
from harness.middleware.danger_detector import DangerCommandDetector
from harness.middleware.feishu_approval import FeishuApprovalMiddleware
from harness.middleware.logger import AuditLogMiddleware

chain = (
    MiddlewareChain()
    .add(AuditLogMiddleware())           # 所有调用都记录日志
    .add(DangerCommandDetector())        # 危险命令检测
)

# 如果启用飞书审批
if feishu_bot and approver_open_id:
    chain.add(FeishuApprovalMiddleware(
        feishu_bot=feishu_bot,
        approver_open_id=approver_open_id,
    ))
```

## 📝 本节小结

| 组件 | 功能 |
|---|---|
| `DangerCommandDetector` | 基于正则识别高危命令，自动拒绝或标记 |
| `FeishuApprovalMiddleware` | 向飞书发送审批卡片，等待人工确认 |
| `MiddlewareChain` | 责任链模式，按序执行 |
| 绝对拒绝 | `rm -rf /`、Fork Bomb 等直接拒绝 |
| 条件审批 | `rm -rf` 等需要人工确认后执行 |

# 09 飞书集成：打通真实世界，go-tiny-claw 接入飞书机器人的事件流

## 🎯 本节目标

将 Agent Harness 与飞书机器人集成，实现通过飞书对话触发 Agent 任务，让 Agent 真正进入真实工作流。

---

## 1. 为什么选择飞书？

企业场景中，飞书（Lark）是理想的 Agent 接入点：

- **事件驱动**：用户在飞书发消息 → 触发 Agent → 回复结果
- **富文本支持**：Markdown、代码块、卡片消息
- **审批流程**：飞书审批 API 可以实现"人工确认"（见第 16 节）
- **群组协作**：Agent 可以在群里被 @ 触发

## 3. 飞书 Bot 配置

### 3.1 在飞书开放平台创建应用

```
1. 访问 https://open.feishu.cn/
2. 创建企业自建应用
3. 开启"机器人"能力
4. 配置消息事件订阅：
   - im.message.receive_v1（接收消息）
5. 申请权限：
   - im:message（发送和读取消息）
   - im:message.group_at_msg（被@时接收消息）
6. 配置 Webhook URL（你的服务器地址）
```

## 4. 飞书集成核心实现

```python
# integrations/feishu/bot.py

import hashlib
import hmac
import json
import time
import asyncio
import logging
from typing import Optional

import httpx
from fastapi import FastAPI, Request, BackgroundTasks

logger = logging.getLogger(__name__)


class FeishuBot:
    """飞书机器人核心封装"""
    
    BASE_URL = "https://open.feishu.cn/open-apis"
    
    def __init__(self, app_id: str, app_secret: str):
        self.app_id = app_id
        self.app_secret = app_secret
        self._access_token: Optional[str] = None
        self._token_expires_at: float = 0
    
    async def get_access_token(self) -> str:
        """获取租户 access_token（带缓存）"""
        now = time.time()
        if self._access_token and now < self._token_expires_at - 60:
            return self._access_token
        
        async with httpx.AsyncClient() as client:
            resp = await client.post(
                f"{self.BASE_URL}/auth/v3/tenant_access_token/internal",
                json={
                    "app_id": self.app_id,
                    "app_secret": self.app_secret,
                }
            )
            data = resp.json()
            self._access_token = data["tenant_access_token"]
            self._token_expires_at = now + data["expire"]
            return self._access_token
    
    async def reply_message(
        self,
        message_id: str,
        content: str,
        msg_type: str = "text",
    ) -> dict:
        """回复消息"""
        token = await self.get_access_token()
        
        if msg_type == "text":
            body = {"text": content}
        elif msg_type == "markdown":
            # 飞书支持 Markdown 富文本
            body = {"zh_cn": {"title": "", "content": [[{"tag": "md", "text": content}]]}}
            msg_type = "post"
        
        async with httpx.AsyncClient() as client:
            resp = await client.post(
                f"{self.BASE_URL}/im/v1/messages/{message_id}/reply",
                headers={"Authorization": f"Bearer {token}"},
                json={
                    "msg_type": msg_type,
                    "content": json.dumps(body, ensure_ascii=False),
                }
            )
            return resp.json()
    
    async def send_message(
        self,
        receive_id: str,
        content: str,
        receive_id_type: str = "open_id",
    ) -> dict:
        """主动发送消息"""
        token = await self.get_access_token()
        
        async with httpx.AsyncClient() as client:
            resp = await client.post(
                f"{self.BASE_URL}/im/v1/messages",
                headers={"Authorization": f"Bearer {token}"},
                params={"receive_id_type": receive_id_type},
                json={
                    "receive_id": receive_id,
                    "msg_type": "text",
                    "content": json.dumps({"text": content}),
                }
            )
            return resp.json()
    
    def verify_signature(
        self,
        timestamp: str,
        nonce: str,
        encrypt_key: str,
        body: str,
        signature: str,
    ) -> bool:
        """验证飞书请求签名"""
        raw = timestamp + nonce + encrypt_key + body
        expected = hashlib.sha256(raw.encode()).hexdigest()
        return hmac.compare_digest(expected, signature)
```

## 5. FastAPI Webhook 服务

```python
# integrations/feishu/webhook.py

import os
from fastapi import FastAPI, Request, BackgroundTasks, HTTPException
from .bot import FeishuBot

app = FastAPI()

bot = FeishuBot(
    app_id=os.environ["FEISHU_APP_ID"],
    app_secret=os.environ["FEISHU_APP_SECRET"],
)

# 处理过的消息 ID（防止重复处理）
processed_messages = set()


@app.post("/webhook")
async def handle_webhook(request: Request, background_tasks: BackgroundTasks):
    """处理飞书推送的 Webhook 事件"""
    body = await request.body()
    data = await request.json()
    
    # 1. 飞书服务验证（首次配置时飞书会发 challenge）
    if "challenge" in data:
        return {"challenge": data["challenge"]}
    
    # 2. 解析事件
    event_type = data.get("header", {}).get("event_type", "")
    
    if event_type == "im.message.receive_v1":
        event = data.get("event", {})
        message = event.get("message", {})
        
        message_id = message.get("message_id")
        
        # 防止重复处理（飞书可能重试推送）
        if message_id in processed_messages:
            return {"msg": "already processed"}
        processed_messages.add(message_id)
        
        # 提取消息文本
        content = json.loads(message.get("content", "{}"))
        text = content.get("text", "").strip()
        
        # 去除 @机器人 前缀
        text = re.sub(r"@\S+\s*", "", text).strip()
        
        if not text:
            return {"msg": "empty message"}
        
        sender_open_id = event.get("sender", {}).get("sender_id", {}).get("open_id")
        
        # 异步处理（不阻塞 webhook 响应）
        background_tasks.add_task(
            process_agent_task,
            text=text,
            message_id=message_id,
            sender_open_id=sender_open_id,
        )
    
    return {"msg": "ok"}


async def process_agent_task(text: str, message_id: str, sender_open_id: str):
    """在后台运行 Agent 任务并回复结果"""
    try:
        # 发送"处理中"提示
        await bot.reply_message(message_id, "⏳ 正在处理，请稍候...")
        
        # 运行 Agent
        from harness import create_default_harness
        harness = create_default_harness()
        result = await harness.run(text)
        
        # 回复结果
        response_text = result.content or "✓ 任务已完成（无文本输出）"
        
        # 如果结果太长，分段发送
        if len(response_text) > 4000:
            chunks = [response_text[i:i+4000] for i in range(0, len(response_text), 4000)]
            for i, chunk in enumerate(chunks):
                prefix = f"（{i+1}/{len(chunks)}）\n" if len(chunks) > 1 else ""
                await bot.reply_message(message_id, prefix + chunk)
        else:
            await bot.reply_message(message_id, response_text)
    
    except Exception as e:
        logger.error(f"Agent 任务执行失败: {e}", exc_info=True)
        await bot.reply_message(message_id, f"❌ 执行出错: {e}")
```

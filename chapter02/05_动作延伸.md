# 05 动作延伸：构建强扩展性的 Tool Registry 与分发机制

## 🎯 本节目标

设计并实现一个支持动态注册、类型安全、易于扩展的 Tool Registry，掌握工具调用的分发机制。

---

## 1. Tool Registry 的核心职责

Tool Registry 是 Agent 的"工具箱"，它需要：

1. **注册**：将工具定义（函数 + 描述 + 参数 Schema）注册到系统
2. **列举**：向 LLM 提供当前可用工具的完整描述
3. **分发**：根据 LLM 的工具调用请求，找到并执行正确的工具函数
4. **容错**：处理工具不存在、参数错误等异常情况

## 2. 工具定义结构

```python
# harness/tools/base.py

from dataclasses import dataclass
from typing import Callable, Any, Optional
import inspect
import json


@dataclass
class ToolDefinition:
    """工具定义"""
    name: str                    # 工具名称（唯一标识）
    description: str             # 功能描述（LLM 用于决策）
    parameters: dict             # JSON Schema 格式的参数定义
    handler: Callable            # 实际执行函数（同步或异步）
    category: str = "general"    # 工具分类（用于分组管理）
    
    def to_llm_schema(self) -> dict:
        """转换为 LLM 可理解的工具描述格式"""
        return {
            "name": self.name,
            "description": self.description,
            "parameters": self.parameters,
        }
```

## 3. Tool Registry 实现

```python
# harness/tools/registry.py

import asyncio
import inspect
import logging
from typing import Any, Callable, Optional
from functools import wraps

from .base import ToolDefinition

logger = logging.getLogger(__name__)


class ToolRegistry:
    """
    工具注册表
    
    支持：
    - 装饰器注册
    - 动态注册
    - 分类管理
    - 并发执行
    """
    
    def __init__(self):
        self._tools: dict[str, ToolDefinition] = {}
    
    # ── 注册方式一：装饰器 ──────────────────────────────────────
    
    def tool(
        self,
        name: str = None,
        description: str = None,
        category: str = "general",
    ):
        """
        装饰器：将函数注册为工具
        
        Usage:
            @registry.tool(description="读取文件内容")
            async def read_file(path: str) -> str:
                with open(path) as f:
                    return f.read()
        """
        def decorator(fn: Callable) -> Callable:
            tool_name = name or fn.__name__
            tool_desc = description or fn.__doc__ or ""
            schema = self._infer_schema(fn)
            
            self.register(ToolDefinition(
                name=tool_name,
                description=tool_desc,
                parameters=schema,
                handler=fn,
                category=category,
            ))
            return fn
        return decorator
    
    # ── 注册方式二：动态注册 ───────────────────────────────────
    
    def register(self, tool: ToolDefinition):
        """动态注册工具"""
        if tool.name in self._tools:
            logger.warning(f"覆盖已存在的工具: {tool.name}")
        self._tools[tool.name] = tool
        logger.info(f"注册工具: {tool.name} [{tool.category}]")
    
    def unregister(self, name: str):
        """注销工具"""
        self._tools.pop(name, None)
    
    # ── 查询 ─────────────────────────────────────────────────
    
    def get_definitions(self, category: str = None) -> list[dict]:
        """
        获取工具定义列表（传给 LLM 的格式）
        
        Args:
            category: 如果指定，只返回该分类的工具
        """
        tools = self._tools.values()
        if category:
            tools = [t for t in tools if t.category == category]
        return [t.to_llm_schema() for t in tools]
    
    def list_tools(self) -> list[str]:
        """列出所有工具名称"""
        return list(self._tools.keys())
    
    # ── 分发执行 ─────────────────────────────────────────────
    
    async def dispatch(self, name: str, args: dict) -> Any:
        """
        根据名称分发并执行工具
        
        Args:
            name: 工具名称
            args: 工具参数（来自 LLM 的 tool_call.args）
            
        Returns:
            工具执行结果（字符串或可序列化对象）
        """
        if name not in self._tools:
            available = ", ".join(self.list_tools())
            raise ToolNotFoundError(
                f"工具 '{name}' 不存在。可用工具: {available}"
            )
        
        tool = self._tools[name]
        
        try:
            # 支持同步和异步函数
            if inspect.iscoroutinefunction(tool.handler):
                result = await tool.handler(**args)
            else:
                result = await asyncio.get_event_loop().run_in_executor(
                    None, lambda: tool.handler(**args)
                )
            return result
        
        except TypeError as e:
            raise ToolArgumentError(f"工具 '{name}' 参数错误: {e}")
        except Exception as e:
            logger.error(f"工具 '{name}' 执行失败: {e}", exc_info=True)
            raise ToolExecutionError(f"工具 '{name}' 执行失败: {e}")
    
    # ── Schema 推断 ──────────────────────────────────────────
    
    def _infer_schema(self, fn: Callable) -> dict:
        """从函数签名自动推断 JSON Schema"""
        sig = inspect.signature(fn)
        hints = fn.__annotations__
        
        properties = {}
        required = []
        
        for param_name, param in sig.parameters.items():
            if param_name == "self":
                continue
            
            # 类型映射
            type_hint = hints.get(param_name, str)
            json_type = {
                str: "string",
                int: "integer",
                float: "number",
                bool: "boolean",
                list: "array",
                dict: "object",
            }.get(type_hint, "string")
            
            properties[param_name] = {"type": json_type}
            
            # 没有默认值 → 必填
            if param.default is inspect.Parameter.empty:
                required.append(param_name)
        
        return {
            "type": "object",
            "properties": properties,
            "required": required,
        }


# 自定义异常
class ToolNotFoundError(Exception): pass
class ToolArgumentError(Exception): pass
class ToolExecutionError(Exception): pass
```

## 4. 实际使用示例

```python
# 创建注册表
registry = ToolRegistry()

# 方式一：装饰器注册
@registry.tool(description="读取指定路径的文件内容", category="filesystem")
async def read_file(path: str) -> str:
    """读取文件内容"""
    with open(path, "r", encoding="utf-8") as f:
        return f.read()


@registry.tool(description="向文件写入内容", category="filesystem")
async def write_file(path: str, content: str) -> str:
    """写入文件"""
    with open(path, "w", encoding="utf-8") as f:
        f.write(content)
    return f"已写入 {len(content)} 字符到 {path}"


@registry.tool(description="在工作目录执行 shell 命令", category="system")
async def bash(command: str) -> str:
    """执行 shell 命令"""
    import subprocess
    result = subprocess.run(
        command, shell=True, capture_output=True, text=True, timeout=30
    )
    output = result.stdout
    if result.returncode != 0:
        output += f"\n[stderr]: {result.stderr}"
    return output


# 方式二：动态注册（适合插件系统）
from harness.tools.base import ToolDefinition

registry.register(ToolDefinition(
    name="http_get",
    description="发起 HTTP GET 请求",
    parameters={
        "type": "object",
        "properties": {
            "url": {"type": "string", "description": "请求 URL"},
            "headers": {"type": "object", "description": "请求头"},
        },
        "required": ["url"],
    },
    handler=my_http_get_function,
    category="network",
))


# 查看已注册工具
print(registry.list_tools())
# ['read_file', 'write_file', 'bash', 'http_get']

# 获取工具描述（传给 LLM）
definitions = registry.get_definitions()

# 按分类获取
fs_tools = registry.get_definitions(category="filesystem")
```

## 5. 工具分组与插件化

```python
# harness/tools/plugins/filesystem.py

def register_filesystem_tools(registry: ToolRegistry):
    """注册文件系统工具集"""
    
    @registry.tool(description="列出目录内容", category="filesystem")
    async def list_dir(path: str = ".") -> str:
        import os
        entries = os.listdir(path)
        return "\n".join(entries)
    
    @registry.tool(description="读取文件", category="filesystem")
    async def read_file(path: str) -> str:
        with open(path, "r", encoding="utf-8") as f:
            return f.read()
    
    @registry.tool(description="写入文件", category="filesystem")
    async def write_file(path: str, content: str) -> str:
        with open(path, "w", encoding="utf-8") as f:
            f.write(content)
        return f"写入成功: {path}"
    
    # 可以注册更多...


# 主程序中按需加载工具集
registry = ToolRegistry()
register_filesystem_tools(registry)

# 如果需要也加载网络工具
# register_network_tools(registry)
```

## 📝 本节小结

| 功能 | 实现方式 |
|---|---|
| 工具注册 | 装饰器 `@registry.tool()` 或 `registry.register()` |
| Schema 推断 | 自动从函数签名提取参数定义 |
| 工具分发 | `registry.dispatch(name, args)` 统一入口 |
| 同步/异步支持 | 自动检测函数类型，在执行器中运行同步函数 |
| 插件化 | `register_xxx_tools(registry)` 按需加载 |

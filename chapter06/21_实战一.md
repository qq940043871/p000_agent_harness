## 3. 完整 CLI 入口实现

```python
# main.py - OpenClaw CLI Engine

import asyncio
import os
import sys
import argparse
import logging
from pathlib import Path

# 配置日志
logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
)
logger = logging.getLogger("openclaw")


def build_harness(workspace: str, model: str = None, verbose: bool = False):
    """构建完整的 Harness 实例"""
    from harness.provider.factory import create_provider, load_provider_from_config
    from harness.tools.registry import ToolRegistry
    from harness.tools.executor import ToolExecutor
    from harness.tools.edit import EditTool
    from harness.tools.plugins.filesystem import register_filesystem_tools
    from harness.prompt.builder import PromptBuilder
    from harness.session.manager import SessionManager
    from harness.context.compaction import ContextCompactor
    from harness.memory.manager import MemoryManager
    from harness.stability.reminders import SystemReminders
    from harness.middleware.base import MiddlewareChain
    from harness.middleware.danger_detector import DangerCommandDetector
    from harness.observability.cost_tracker import CostTracker, TrackedProvider
    from harness.observability.tracing import Tracer
    from harness.main_loop import MainLoop
    
    # 1. 创建 Provider
    provider = load_provider_from_config(
        config_path=os.path.join(workspace, "config.yaml"),
        provider_name=model,
    )
    
    # 2. 构建工具注册表
    registry = ToolRegistry()
    register_filesystem_tools(registry, workspace)
    
    # 注册记忆工具
    memory = MemoryManager(workspace)
    from harness.memory.todos import TodoManager
    todos = TodoManager(workspace)
    
    @registry.tool(description="将重要信息保存到持久化记忆库", category="memory")
    async def remember(section: str, content: str) -> str:
        return memory.remember(section, content)
    
    @registry.tool(description="读取持久化记忆库的内容", category="memory")
    async def recall(query: str = "") -> str:
        return memory.recall(query or None)
    
    # 3. 构建 Middleware 链
    middleware = MiddlewareChain().add(DangerCommandDetector())
    
    # 4. 构建 Prompt
    prompt = (
        PromptBuilder(workspace)
        .load_identity()
        .load_agents_md()
        .load_memory()
        .load_skills()
        .inject_working_directory()
        .inject_datetime()
        .build()
    )
    
    # 5. 构建 Main Loop
    loop = MainLoop(
        provider=provider,
        tool_registry=registry,
        system_prompt=prompt,
        max_turns=50,
        max_context_tokens=150000,
        middleware=middleware,
    )
    
    return loop, SessionManager(workspace)


async def run_task(task: str, workspace: str = ".", **kwargs):
    """执行单个任务"""
    loop, session_manager = build_harness(workspace, **kwargs)
    
    session = session_manager.new_session(task[:50])
    print(f"\n🚀 OpenClaw 开始执行任务")
    print(f"   Session: {session.session_id}")
    print(f"   工作目录: {workspace}")
    print(f"   任务: {task}")
    print(f"{'─'*60}\n")
    
    result = await loop.run(task)
    
    print(f"\n{'─'*60}")
    print(f"✅ 任务完成（{result.stats.turns} 轮 / {result.stats.total_tokens:,} tokens）")
    print(f"\n📝 结果:\n{result.content}")
    
    return result


async def interactive_mode(workspace: str = "."):
    """交互模式：持续接受用户输入"""
    loop, session_manager = build_harness(workspace)
    session = session_manager.new_session("interactive")
    
    print(f"\n🦀 OpenClaw 交互模式（输入 'exit' 退出）")
    print(f"   工作目录: {Path(workspace).absolute()}\n")
    
    while True:
        try:
            task = input(">> ").strip()
        except (EOFError, KeyboardInterrupt):
            break
        
        if task.lower() in ("exit", "quit", "q"):
            break
        
        if not task:
            continue
        
        result = await loop.run(task)
        print(f"\n{result.content}\n")


def main():
    parser = argparse.ArgumentParser(description="OpenClaw Agent CLI")
    parser.add_argument("task", nargs="?", help="要执行的任务（不提供则进入交互模式）")
    parser.add_argument("-w", "--workspace", default=".", help="工作目录")
    parser.add_argument("-m", "--model", help="使用的模型（覆盖配置文件）")
    parser.add_argument("-v", "--verbose", action="store_true", help="详细日志")
    
    args = parser.parse_args()
    
    if args.verbose:
        logging.getLogger().setLevel(logging.DEBUG)
    
    if args.task:
        asyncio.run(run_task(
            task=args.task,
            workspace=args.workspace,
            model=args.model,
            verbose=args.verbose,
        ))
    else:
        asyncio.run(interactive_mode(workspace=args.workspace))


if __name__ == "__main__":
    main()
```

## 4. 实战演示：Bug 修复任务

```bash
# 创建一个有 Bug 的测试项目
cat > /tmp/test_project/buggy.py << 'EOF'
def calculate_average(numbers):
    total = 0
    for n in numbers:
        total += n
    return total / len(numbers)  # Bug: 未处理空列表

result = calculate_average([])
print(result)
EOF

# 运行 OpenClaw 修复 Bug
python main.py "修复 /tmp/test_project/buggy.py 中的 Bug，使其能正确处理空列表，并添加相应的测试" \
  -w /tmp/test_project \
  -v
```

**预期执行轨迹：**

```
[Turn 1] 读取 buggy.py，分析代码
[Turn 2] 识别 Bug：空列表除以 0

[Turn 4] 创建测试文件
[Turn 5] 运行测试验证
→ 完成！
```
